2. ПРОВЕРКА СИСТЕМЫ (после подключения)
================================================================================

hostname

nvidia-smi
ollama --version

================================================================================
3. УСТАНОВКА OLLAMA (если не установлен)
================================================================================

curl -fsSL https://ollama.com/install.sh | sh

sudo mkdir -p /etc/systemd/system/ollama.service.d

echo '[Service]' | sudo tee /etc/systemd/system/ollama.service.d/override.conf

echo 'Environment="OLLAMA_HOST=0.0.0.0:11434"' | sudo tee -a /etc/systemd/system/ollama.service.d/override.conf

sudo systemctl daemon-reload
sudo systemctl restart ollama
sudo systemctl enable ollama

systemctl status ollama

================================================================================
4. УСТАНОВКА NVIDIA ДРАЙВЕРОВ (если не установлены)
================================================================================

sudo apt update
sudo apt install -y nvidia-driver-535 nvidia-utils-535

ЗАТЕМ ПЕРЕЗАГРУЗКА:
sudo reboot

================================================================================
5. ПОСЛЕ ПЕРЕЗАГРУЗКИ - ПРОВЕРКА GPU
================================================================================

ssh -p 42 user@YOUR_PUBLIC_HOST

nvidia-smi

Должна отобразиться NVIDIA A40

================================================================================
6. НАСТРОЙКА ФАЙРВОЛА
================================================================================

sudo apt install -y ufw
sudo ufw allow 22/tcp
sudo ufw allow 11434/tcp
sudo ufw --force enable
sudo ufw status

================================================================================

ollama list
