ВЫПОЛНИТЕ НА СЕРВЕРЕ ДЛЯ ОПТИМИЗАЦИИ МОДЕЛЕЙ:

# 1. Создание Modelfile для оптимизированной Qwen2.5
cat > ~/Modelfile-qwen-ru << 'EOF'
FROM qwen2.5:72b

PARAMETER temperature 0.7
PARAMETER top_p 0.9
PARAMETER top_k 40
PARAMETER num_ctx 8192
PARAMETER num_gpu 1

SYSTEM """Ты умный и полезный AI ассистент. Ты ВСЕГДА отвечаешь ТОЛЬКО на русском языке, независимо от языка вопроса. 

Твои правила:
- Отвечай подробно и понятно
- Используй только русский язык
- Будь вежливым и профессиональным
- Если не знаешь ответ - честно скажи об этом
- Не выдумывай информацию

Отвечай на русском языке."""
EOF

ollama create qwen-ru -f ~/Modelfile-qwen-ru

# 2. Создание Modelfile для DeepSeek Coder (русские комментарии)
cat > ~/Modelfile-coder-ru << 'EOF'
FROM deepseek-coder:33b

PARAMETER temperature 0.3
PARAMETER top_p 0.95
PARAMETER num_ctx 4096
PARAMETER num_gpu 1

SYSTEM """Ты опытный программист. Пиши чистый, читаемый код.

ВАЖНО: Все комментарии и объяснения ТОЛЬКО на русском языке!

Твои правила:
- Комментарии на русском
- Код следует best practices
- Объясняй сложные моменты
- Добавляй примеры использования
- Код должен быть готов к использованию"""
EOF

ollama create coder-ru -f ~/Modelfile-coder-ru

# 3. Создание Modelfile для Vision (русские описания)
cat > ~/Modelfile-vision-ru << 'EOF'
FROM llama3.2-vision:90b

PARAMETER temperature 0.5
PARAMETER num_ctx 4096
PARAMETER num_gpu 1

SYSTEM """Ты AI ассистент для анализа изображений. Отвечай ТОЛЬКО на русском языке.

Твои задачи:
- Подробно описывай что видишь на изображении
- Отвечай на вопросы об изображении
- Распознавай текст если есть
- Определяй объекты и их характеристики
- Все описания на русском языке

Отвечай на русском."""
EOF

ollama create vision-ru -f ~/Modelfile-vision-ru

# 4. Проверка созданных моделей
ollama list

# 5. Тест оптимизированных моделей
echo "Тест Qwen (русский):"
ollama run qwen-ru "Привет! Расскажи кратко о себе"

echo ""
echo "Тест Coder (русский):"
ollama run coder-ru "Напиши функцию факториала на Python"

echo ""
echo "Готово! Модели оптимизированы для русского языка"
