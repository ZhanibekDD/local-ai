================================================================================
КОМАНДЫ ДЛЯ ВЫПОЛНЕНИЯ В ТЕРМИНАЛЕ (вы уже подключены к серверу)
================================================================================

Скопируйте и выполните команды по порядку:

================================================================================
1. ПРОВЕРКА OLLAMA
================================================================================

ollama --version

systemctl status ollama

================================================================================
2. ЕСЛИ OLLAMA НЕ УСТАНОВЛЕН - УСТАНОВКА
================================================================================

curl -fsSL https://ollama.com/install.sh | sh

sudo mkdir -p /etc/systemd/system/ollama.service.d

echo '[Service]' | sudo tee /etc/systemd/system/ollama.service.d/override.conf

echo 'Environment="OLLAMA_HOST=0.0.0.0:11434"' | sudo tee -a /etc/systemd/system/ollama.service.d/override.conf

sudo systemctl daemon-reload
sudo systemctl restart ollama
sudo systemctl enable ollama

systemctl status ollama

================================================================================
3. УСТАНОВКА NVIDIA ДРАЙВЕРОВ
================================================================================

sudo apt update

sudo apt install -y nvidia-driver-535 nvidia-utils-535

================================================================================
4. ПЕРЕЗАГРУЗКА (ОБЯЗАТЕЛЬНО!)
================================================================================

sudo reboot

================================================================================
5. ПОСЛЕ ПЕРЕЗАГРУЗКИ - ПОДКЛЮЧИТЕСЬ СНОВА
================================================================================

ssh -p 42 user@YOUR_PUBLIC_HOST

Пароль: REDACTED

================================================================================
6. ПРОВЕРКА GPU
================================================================================

nvidia-smi

Вы должны увидеть NVIDIA A40 48GB

================================================================================
7. НАСТРОЙКА ФАЙРВОЛА
================================================================================

sudo apt install -y ufw
sudo ufw allow 22/tcp
sudo ufw allow 11434/tcp
sudo ufw --force enable
sudo ufw status

================================================================================
8. ЗАГРУЗКА МОДЕЛЕЙ (30-60 минут!)
================================================================================

ollama list

ollama pull qwen2.5:72b

ollama pull llama3.2-vision:90b

ollama pull deepseek-coder:33b

ollama list

================================================================================
9. ТЕСТИРОВАНИЕ
================================================================================

ollama run qwen2.5:72b "Напиши Hello World на Python"

curl http://localhost:11434/api/generate -d '{"model":"qwen2.5:72b","prompt":"Привет!","stream":false}'
