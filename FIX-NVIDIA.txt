================================================================================
ИСПРАВЛЕНИЕ NVIDIA ДРАЙВЕРОВ
================================================================================

Скопируйте и выполните команды по порядку в вашем открытом терминале:

================================================================================
1. ИСПРАВЛЕНИЕ ЗАВИСИМОСТЕЙ
================================================================================

sudo apt --fix-broken install

sudo apt update

================================================================================
2. ОЧИСТКА СТАРЫХ ДРАЙВЕРОВ
================================================================================

sudo apt autoremove -y

sudo apt autoclean

================================================================================
3. УСТАНОВКА NVIDIA ДРАЙВЕРОВ (правильный способ)
================================================================================

sudo apt install -y nvidia-driver-535

================================================================================
4. ЕСЛИ ОШИБКА - АЛЬТЕРНАТИВНЫЙ СПОСОБ
================================================================================

sudo apt purge -y nvidia-* libnvidia-*

sudo apt autoremove -y

sudo apt update

sudo ubuntu-drivers install

================================================================================
5. ПЕРЕЗАГРУЗКА
================================================================================

sudo reboot

================================================================================
6. ПОСЛЕ ПЕРЕЗАГРУЗКИ - ПРОВЕРКА
================================================================================

ssh -p 42 user@YOUR_PUBLIC_HOST

nvidia-smi

================================================================================
7. ЕСЛИ GPU РАБОТАЕТ - УСТАНОВКА OLLAMA
================================================================================

curl -fsSL https://ollama.com/install.sh | sh

sudo mkdir -p /etc/systemd/system/ollama.service.d

echo '[Service]' | sudo tee /etc/systemd/system/ollama.service.d/override.conf

echo 'Environment="OLLAMA_HOST=0.0.0.0:11434"' | sudo tee -a /etc/systemd/system/ollama.service.d/override.conf

sudo systemctl daemon-reload
sudo systemctl restart ollama
sudo systemctl enable ollama

systemctl status ollama

================================================================================
8. ЗАГРУЗКА МОДЕЛЕЙ
================================================================================

ollama pull qwen2.5:72b

ollama pull llama3.2-vision:90b

ollama pull deepseek-coder:33b

ollama list

================================================================================
9. ТЕСТ
================================================================================

ollama run qwen2.5:72b "Напиши Hello World на Python"
