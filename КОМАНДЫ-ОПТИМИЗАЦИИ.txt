СКОПИРУЙТЕ ЭТИ КОМАНДЫ В SSH ТЕРМИНАЛ НА СЕРВЕРЕ:

cat > ~/optimize.sh << 'SCRIPTEND'
#!/bin/bash
echo "Оптимизация моделей..."

cat > /tmp/qwen-ru << 'EOF'
FROM qwen2.5:72b
PARAMETER temperature 0.7
PARAMETER num_ctx 8192
PARAMETER num_gpu 1
SYSTEM "Ты AI ассистент. Отвечай ТОЛЬКО на русском языке, даже если вопрос на английском."
EOF

ollama create qwen-ru -f /tmp/qwen-ru

cat > /tmp/coder-ru << 'EOF'
FROM deepseek-coder:33b
PARAMETER temperature 0.3
PARAMETER num_ctx 4096
PARAMETER num_gpu 1
SYSTEM "Ты программист. Все комментарии ТОЛЬКО на русском языке."
EOF

ollama create coder-ru -f /tmp/coder-ru

cat > /tmp/vision-ru << 'EOF'
FROM llama3.2-vision:90b
PARAMETER temperature 0.5
PARAMETER num_ctx 4096
PARAMETER num_gpu 1
SYSTEM "Ты анализируешь изображения. Отвечай ТОЛЬКО на русском языке."
EOF

ollama create vision-ru -f /tmp/vision-ru

sudo systemctl stop ollama

sudo mkdir -p /etc/systemd/system/ollama.service.d

sudo tee /etc/systemd/system/ollama.service.d/performance.conf > /dev/null << 'EOF'
[Service]
Environment="OLLAMA_NUM_PARALLEL=2"
Environment="OLLAMA_MAX_LOADED_MODELS=1"
Environment="OLLAMA_FLASH_ATTENTION=1"
EOF

sudo systemctl daemon-reload
sudo systemctl start ollama

echo "Готово!"
ollama list
SCRIPTEND

chmod +x ~/optimize.sh

bash ~/optimize.sh
