БЫСТРАЯ НАСТРОЙКА ДЛЯ УСКОРЕНИЯ (выполните на сервере):

sudo systemctl stop ollama

sudo tee /etc/systemd/system/ollama.service.d/override.conf > /dev/null << 'EOF'
[Service]
Environment="OLLAMA_HOST=0.0.0.0:11434"
Environment="OLLAMA_NUM_PARALLEL=2"
Environment="OLLAMA_MAX_LOADED_MODELS=1"
Environment="OLLAMA_FLASH_ATTENTION=1"
Environment="OLLAMA_KV_CACHE_TYPE=q8_0"
EOF

sudo systemctl daemon-reload

sudo systemctl start ollama

cat > ~/Modelfile-qwen-ru << 'EOF'
FROM qwen2.5:72b
PARAMETER temperature 0.7
PARAMETER num_ctx 8192
PARAMETER num_gpu 1
SYSTEM """Ты AI ассистент. Отвечай ТОЛЬКО на русском языке. Будь полезным и понятным."""
EOF

ollama create qwen-ru -f ~/Modelfile-qwen-ru

cat > ~/Modelfile-coder-ru << 'EOF'
FROM deepseek-coder:33b
PARAMETER temperature 0.3
PARAMETER num_ctx 4096
PARAMETER num_gpu 1
SYSTEM """Ты программист. Все комментарии ТОЛЬКО на русском языке."""
EOF

ollama create coder-ru -f ~/Modelfile-coder-ru

cat > ~/Modelfile-vision-ru << 'EOF'
FROM llama3.2-vision:90b
PARAMETER temperature 0.5
PARAMETER num_ctx 4096
PARAMETER num_gpu 1
SYSTEM """Ты анализируешь изображения. Отвечай ТОЛЬКО на русском языке."""
EOF

ollama create vision-ru -f ~/Modelfile-vision-ru

ollama list

echo "Готово! Перезапустите бота"
